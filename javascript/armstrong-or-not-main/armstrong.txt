<html>
<head>
    <title>Armstrong Number Checker</title>
    <script>
        function armstrong() {
            var rem;
            var res = 0; 
            var num1 = parseInt(document.getElementById('N1').value);
            var dummy = num1;
            while (dummy != 0) { 
                rem = dummy % 10;
                res = res + rem * rem * rem;
                dummy = parseInt(dummy / 10);
            }
            if (res == num1) {
                document.getElementById('res').innerHTML = num1 + " is an Armstrong number";
            } else {
                document.getElementById('res').innerHTML = num1 + " is not an Armstrong number";
            }
        }
    </script>
</head>
<body>
    <label for="number">Enter a number :</label>
    <input type="number" id="N1"><br><br>
    <input type="button" onclick="armstrong()" value="Check Armstrong">
    <br><br>
    <div id="res"></div>
</body>
</html>
