            Prime or not..

        <script>
            function prime(){
                var num=parseInt(document.getElementById('n1').value);
                var flag=0;
                var i;
                if(num==0 || num==1)
                {
                    flag=1;
                }
                for(i=2;i<=num/2;i++)
                {
                    if (num % i == 0) 
                    {
                    flag = 1;
                    break;
                    }
                }
                if(flag==0){
                    document.getElementById('res').innerHTML=num + " is a prime number...";
                }
                else{
                    document.getElementById('res').innerHTML=num + " is not a prime number...";
                }
            }
        </script>
